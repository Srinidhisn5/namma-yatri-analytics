üíª SQL Analysis Queries Used

Query 1: Peak Hours Analysis

SELECT EXTRACT(HOUR FROM ride_start_time) AS hour_of_day, COUNT(*) AS total_rides, AVG(fare_amount) AS avg_fare, AVG(trip_duration_minutes) AS avg_duration FROM rides WHERE ride_date >= CURRENT_DATE - INTERVAL '30 days' AND ride_status = 'completed' GROUP BY hour_of_day ORDER BY total_rides DESC LIMIT 10;

Query 2: Driver Performance Metrics

WITH driver_stats AS ( SELECT driver_id, COUNT(*) AS total_trips, AVG(rating) AS avg_rating, SUM(fare_amount) AS total_earnings, AVG(acceptance_rate) AS avg_acceptance FROM rides r JOIN drivers d ON r.driver_id = d.id WHERE ride_date >= CURRENT_DATE - INTERVAL '30 days' GROUP BY driver_id ) SELECT CASE WHEN total_trips >= 20 THEN 'High Performer' WHEN total_trips >= 10 THEN 'Medium Performer' ELSE 'Low Performer' END AS performance_tier, COUNT(*) AS driver_count, AVG(total_earnings) AS avg_earnings, AVG(avg_rating) AS avg_rating FROM driver_stats GROUP BY performance_tier;

Query 3: Route Optimization Analysis

SELECT pickup_location, dropoff_location, COUNT(*) AS route_frequency, AVG(trip_duration_minutes) AS avg_duration, AVG(distance_km) AS avg_distance, AVG(fare_amount) AS avg_fare, AVG(fare_amount / distance_km) AS fare_per_km FROM rides WHERE ride_status = 'completed' AND ride_date >= CURRENT_DATE - INTERVAL '30 days' GROUP BY pickup_location, dropoff_location HAVING COUNT(*) >= 50 ORDER BY route_frequency DESC LIMIT 15;
